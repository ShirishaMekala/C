#include <stdio.h>

int main()
{
    int month=3;
    switch(month)
    {
        case 1:
        printf("january");
        break;
        case 2:
        printf("february");
        break;
        case 3:
        printf("march");
        break;
        case 4:
        printf("april");
        break;
        case 5:
        printf("may");
        break;
        case 6:
        printf("june");
        break;
    }
}
